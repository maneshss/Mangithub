def evaluate(model, X_train, X_test, y_train, y_test):
    pred_tr = np.expm1(model.predict(X_train))
    pred_te = np.expm1(model.predict(X_test))
    return {
        "Train RMSE": float(mean_squared_error(y_train, pred_tr, squared=False)),
        "Test RMSE":  float(mean_squared_error(y_test,  pred_te, squared=False)),
        "Train R2":   float(r2_score(y_train, pred_tr)),
        "Test R2":    float(r2_score(y_test,  pred_te)),
        "Pred test min": float(np.min(pred_te)),
        "Pred test max": float(np.max(pred_te)),
    }

# Model 2: enhanced regression (polynomial and piecewise)

def add_engineered_features(X):
    X = X.copy()

    # --- Polynomial (like age_sq) ---
    X["house_age"] = X["YrSold"] - X["YearBuilt"]
    X["house_age_sq"] = X["house_age"] ** 2

    # --- Piecewise / threshold (like bmi_above_30) ---
    # Extra penalty once the house is "old"
    X["age_above_50"] = (X["house_age"] - 50).clip(lower=0)

    # --- Total area feature (very strong) ---
    X["TotalSF"] = X["TotalBsmtSF"] + X["1stFlrSF"] + X["2ndFlrSF"]
    X["TotalSF_sq"] = X["TotalSF"] ** 2

    # --- Interaction example (quality  size) ---
    # (works well in linear models)
    X["Qual_x_TotalSF"] = X["OverallQual"] * X["TotalSF"]

    # --- Bathrooms (more realistic than separate columns) ---
    X["TotalBath"] = (
        X["FullBath"] + 0.5 * X["HalfBath"] +
        X["BsmtFullBath"] + 0.5 * X["BsmtHalfBath"]
    )

    # --- Piecewise remodeling recency ---
    X["remod_age"] = X["YrSold"] - X["YearRemodAdd"]
    X["remod_recent_bonus"] = (10 - X["remod_age"]).clip(lower=0)  # 0 if remod >10 yrs ago

    return X

from sklearn.model_selection import KFold
from sklearn.model_selection import cross_val_score

kf = KFold(n_splits = 5, shuffle = True, random_state = 42)

def cross_validate_model(X, y, enhanced = False):
  rmse_list, r2_list = [], []

  for train_index, test_index in kf.split(X):
    X_train, X_test = X.iloc[train_index], X.iloc[test_index]
    y_train, y_test = y.iloc[train_index], y.iloc[test_index]

def cross_validate_model(X, y, enhanced=False, n_splits=5, random_state=42):
    kf = KFold(n_splits=n_splits, shuffle=True, random_state=random_state)

    rmse_list = []
    r2_list = []

    for train_idx, test_idx in kf.split(X):
        X_train = X.iloc[train_idx].copy()
        X_test  = X.iloc[test_idx].copy()
        y_train = y.iloc[train_idx]
        y_test  = y.iloc[test_idx]

        if enhanced:
            X_train = add_engineered_features(X_train)
            X_test  = add_engineered_features(X_test)

        # --- ADD THESE LINES (minimal change) ---
        X_train = X_train.select_dtypes(include=[np.number]).replace([np.inf, -np.inf], np.nan)
        X_test  = X_test.select_dtypes(include=[np.number]).replace([np.inf, -np.inf], np.nan)

        imputer = SimpleImputer(strategy="median")
        X_train = imputer.fit_transform(X_train)
        X_test  = imputer.transform(X_test)
        # --- END ADD ---

        model = LinearRegression()
        model.fit(X_train, y_train)

        y_pred = model.predict(X_test)

        rmse_list.append(mean_squared_error(y_test, y_pred, squared=False))
        r2_list.append(r2_score(y_test, y_pred))

    return {
        "CV RMSE": float(np.mean(rmse_list)),
        "CV R2": float(np.mean(r2_list))

What the models tell us in laymans terms

Both of my models are trying to answer a simple question: what makes one house more expensive than another? The results are consistent and intuitive.

Baseline model (basic regression): The baseline model shows that house prices are largely driven by the size and overall condition of the house. In plain terms, homes with larger spaces (bigger floor area, basement area, garage capacity) and better quality ratings generally sell for more. However, the baseline model tends to be less accurate for very expensive homes because high-end prices can be influenced by premium factors that dont increase in a perfectly straight line.

Enhanced regression model (engineered features: polynomial + piecewise): The enhanced model improves accuracy because it better matches how real estate pricing works in the real world:

1.Bigger homes sell for more, but not in a completely linear way. Adding a square-sized term helps capture the idea that the value from extra square footage changes as homes get larger. 2.Age has a threshold effect. A home that is older than a certain point can behave differently in price due to wear, renovations, or style. The piecewise age feature captures that change. 3,Size and quality work together. A large home with high quality commands a higher premium than one with average quality. The interaction-style engineered features help reflect this.

Because of these improvements, the enhanced model performed better on the test set:

1.Baseline Test RMSE 24,602 vs Enhanced Test RMSE 23,614 (lower is better) 2.Baseline Test R 0.913 vs Enhanced Test R 0.920 (higher is better)

Why using log(SalePrice) helped: House prices are not evenly distributedthere are a few very expensive homes that can distort the model. Using a log transformation makes the model focus more on relative (percentage) differences instead of being dominated by a few large dollar amounts. This makes predictions more stable and increases generalization capability.

Overall conclusion: In everyday language, the models say that price increases with space, quality, and location-related attributes, but the relationships are not perfectly linear. The enhanced model performs better because it captures realistic effects, such as changing returns to extra space and threshold behavior in older homes.
